	]}

	$scope.refreshData = function(){
		var newDatas = [];
		angular.forEach($scope.jstabledata.rows, function(v){
			v.name = '用户' + new Date().getTime();
			newDatas.push(v);
		})

		$scope.jstabledata = {
			total: 1000,
			rows: newDatas
		}
	}

}])
